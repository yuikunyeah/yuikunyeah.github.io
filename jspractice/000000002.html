switchとは

例のコードを見てください(ファイル名:sample.js)

例

function gate(){
var sample = prompt("1~3を入力");
switch(sample)//変数名{
//もし答えが1なら
case 1:
alert('パート1です')
break;
//もし答えが2なら
case 2:
alert('パート2です')
break;
//もし答えが3なら
case 3:
alert('パート3です')
break;
default{ //1~3ではないなら
alert('値を正確に入力してください');
}
}
}
